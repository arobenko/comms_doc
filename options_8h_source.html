73 template<std::intmax_t TMinValue, std::intmax_t TMaxValue>
74 struct NumValueRangeValidator
75 {
76  static_assert(
77  TMinValue <= TMaxValue,
78  "Min value must be not greater than Max value");
79 
80  template <typename TField>
81  constexpr bool operator()(const TField& field) const
82  {
83  using MinTag = typename std::conditional<
84  (std::numeric_limits<decltype(MinValue)>::min() < MinValue),
85  CompareTag,
86  ReturnTrueTag
87  >::type;
88 
89  using MaxTag = typename std::conditional<
90  (MaxValue < std::numeric_limits<decltype(MaxValue)>::max()),
91  CompareTag,
92  ReturnTrueTag
93  >::type;
94 
95  return aboveMin(field.value(), MinTag()) && belowMax(field.value(), MaxTag());
96  }
97 
98 private:
99  struct ReturnTrueTag {};
100  struct CompareTag {};
101 
102  template <typename TValue>
103  static constexpr bool aboveMin(const TValue& value, CompareTag)
104  {
105  using ValueType = typename std::decay<decltype(value)>::type;
106  return (static_cast<ValueType>(MinValue) <= static_cast<ValueType>(value));
107  }
108 
109  template <typename TValue>
110  static constexpr bool aboveMin(const TValue&, ReturnTrueTag)
111  {
112  return true;
113  }
114 
115  template <typename TValue>
116  static constexpr bool belowMax(const TValue& value, CompareTag)
117  {
118  using ValueType = typename std::decay<decltype(value)>::type;
119  return (value <= static_cast<ValueType>(MaxValue));
120  }
121 
122  template <typename TValue>
123  static constexpr bool belowMax(const TValue&, ReturnTrueTag)
124  {
125  return true;
126  }
127 
128 
129  static const auto MinValue = TMinValue;
130  static const auto MaxValue = TMaxValue;
131 };
