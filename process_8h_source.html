61 template <typename TBufIter, typename TFrame, typename TMsg, typename... TExtraValues>
62 comms::ErrorStatus processSingle(
63  TBufIter& bufIter,
64  std::size_t len,
65  TFrame&& frame,
66  TMsg& msg,
67  TExtraValues... extraValues)
68 {
69  std::size_t consumed = 0U;
70  auto onExit =
71  comms::util::makeScopeGuard(
72  [&bufIter, &consumed]()
73  {
74  std::advance(bufIter, consumed);
75  });
76  static_cast<void>(onExit);
77 
78  while (consumed < len) {
79  auto begIter = comms::readIteratorFor(msg, bufIter + consumed);
80  auto iter = begIter;
81 
82  // Do the read
83  auto es = frame.read(msg, iter, len - consumed, extraValues...);
84  if (es == comms::ErrorStatus::NotEnoughData) {
85  return es;
86  }
87 
88  if (es == comms::ErrorStatus::ProtocolError) {
89  // Something is not right with the data, remove one character and try again
90  ++consumed;
91  continue;
92  }
93 
94  consumed += std::distance(begIter, iter);
95  return es;
96  }
97 
98  return comms::ErrorStatus::NotEnoughData;
99 }

255 template <typename TBufIter, typename TFrame, typename THandler, typename... TExtraValues>
256 std::size_t processAllWithDispatch(

258  std::size_t len,
259  TFrame&& frame,
260  THandler& handler)
261 {
262  std::size_t consumed = 0U;
263  using FrameType = typename std::decay<decltype(frame)>::type;
264  using MsgPtr = typename FrameType::MsgPtr;
265  while (consumed < len) {
266  auto begIter = bufIter + consumed;
267  auto iter = begIter;
268 
269  MsgPtr msg;
270  auto es = processSingleWithDispatch(iter, len - consumed, std::forward<TFrame>(frame), msg, handler);
271  consumed += std::distance(begIter, iter);
272  if (es == comms::ErrorStatus::NotEnoughData) {
273  break;
274  }
275  COMMS_ASSERT(consumed <= len);
276  }
277 
278  return consumed;
