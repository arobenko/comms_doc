114  template <typename TRet, typename TRatio, typename TField>
115  static TRet getValueInternal(const TField& field, ConvertToIntTag)
116  {
117  static_assert(std::is_integral<TRet>::value,
118  "TRet is expected to be integral type");
119

121  using ValueType = typename FieldType::ValueType;
122 
123  static_assert(
124  std::is_integral<ValueType>::value ||

126  std::is_enum<ValueType>::value,
127  "Unexpected field in units conversion");

201  template <typename TRatio, typename TField, typename TVal>
202  static void setValueInternal(TField& field, TVal&& value, ConvertToFpTag)
203  {
204  using DecayedType = typename std::decay<decltype(value)>::type;
205  using FieldType = typename std::decay<decltype(field)>::type;
206  using ValueType = typename FieldType::ValueType;
207 
208  auto epsilon = DecayedType(0);
209  if ((TRatio::num < TRatio::den) && std::is_integral<ValueType>::value) {
210  epsilon = static_cast<DecayedType>(TRatio::num) / static_cast<DecayedType>(TRatio::den + 1);
211  }
212 
213  if (epsilon < DecayedType(0)) {
214  epsilon = -epsilon;
215  }
216 
217  if (value < DecayedType(0)) {
218  epsilon = -epsilon;
219  }
220 
221  field.value() =
222  static_cast<ValueType>(
223  ((value + epsilon) * static_cast<DecayedType>(TRatio::den)) / static_cast<DecayedType>(TRatio::num));
224  }
