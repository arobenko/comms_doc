124  template <typename TRet, typename TRatio, typename TField>
125  static TRet getValueInternal(const TField& field, ConvertToIntTag)
126  {
127  static_assert(std::is_integral<TRet>::value,
128  "TRet is expected to be integral type");
129 
130  using FieldType = typename std::decay<decltype(field)>::type;
131  using ValueType = typename FieldType::ValueType;
132 
133  static_assert(
134  std::is_integral<ValueType>::value ||
135  std::is_floating_point<ValueType>::value ||
136  std::is_enum<ValueType>::value,
137  "Unexpected field in units conversion");
138 
139  using CastType =
140  typename std::conditional<
141  std::is_floating_point<ValueType>::value,
142  typename std::conditional<
143  std::is_same<ValueType, float>::value,
144  double,
145  ValueType
146  >::type,
147  typename std::conditional<
148  std::is_signed<TRet>::value,
149  std::intmax_t,
150  std::uintmax_t
151  >::type
152  >::type;
153

211  template <typename TRatio, typename TField, typename TVal>
212  static void setValueInternal(TField& field, TVal&& value, ConvertToFpTag)
213  {
214  using DecayedType = typename std::decay<decltype(value)>::type;
215  using FieldType = typename std::decay<decltype(field)>::type;
216  using ValueType = typename FieldType::ValueType;
217 
218  auto epsilon = DecayedType(0);
219  if ((TRatio::num < TRatio::den) && std::is_integral<ValueType>::value) {
220  epsilon = static_cast<DecayedType>(TRatio::num) / static_cast<DecayedType>(TRatio::den + 1);
221  }
222 
223  if (epsilon < DecayedType(0)) {
224  epsilon = -epsilon;
225  }
226 
227  if (value < DecayedType(0)) {
228  epsilon = -epsilon;
229  }
230 
231  field.value() =
232  static_cast<ValueType>(
233  ((value + epsilon) * static_cast<DecayedType>(TRatio::den)) / static_cast<DecayedType>(TRatio::num));
234  }
