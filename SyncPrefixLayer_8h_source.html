99  template <typename TMsg, typename TIter, typename TNextLayerReader, typename... TExtraValues>
100  comms::ErrorStatus doRead(
101  Field& field,
102  TMsg& msg,
103  TIter& iter,
104  std::size_t size,
105  TNextLayerReader&& nextLayerReader,
106  TExtraValues... extraValues)
107  {
108  auto beforeReadIter = iter;
109  auto es = field.read(iter, size);
110  if (es == comms::ErrorStatus::NotEnoughData) {
111  BaseImpl::updateMissingSize(field, size, extraValues...);
112  }
113 
114  if (es != comms::ErrorStatus::Success) {
115  return es;
116  }
117 
118  if (field != Field()) {
119  // doesn't match expected
120  return comms::ErrorStatus::ProtocolError;
121  }
122 
123  auto fieldLen = static_cast<std::size_t>(std::distance(beforeReadIter, iter));
124  return nextLayerReader.read(msg, iter, size - fieldLen, extraValues...);
125  }
126 
144  template <typename TMsg, typename TIter, typename TNextLayerWriter>
145  comms::ErrorStatus doWrite(
146  Field& field,
147  const TMsg& msg,
148  TIter& iter,
149  std::size_t size,
150  TNextLayerWriter&& nextLayerWriter) const
151  {
152  auto es = field.write(iter, size);
153  if (es != ErrorStatus::Success) {
154  return es;
155  }
156 
157  COMMS_ASSERT(field.length() <= size);
158  return nextLayerWriter.write(msg, iter, size - field.length());
159  }
