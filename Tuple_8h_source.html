706 template <std::size_t TFromIdx, std::size_t TToIdx, std::size_t TCount>
707 class TupleSelectedTypeHelper
708 {
709  static_assert(TCount == (TToIdx - TFromIdx), "Internal error: Bad parameters");
710  static_assert(TFromIdx < TToIdx, "Internal error: Bad parameters");
711 public:
712  template <typename TTuple, typename TFunc>
713  static void exec(std::size_t idx, TFunc&& func)
714  {
715  using Tuple = typename std::decay<TTuple>::type;
716  static_assert(IsTuple<Tuple>::Value, "TTuple must be std::tuple");
717  static const std::size_t TupleSize = std::tuple_size<Tuple>::value;
718  static_assert(TCount <= TupleSize, "Incorrect TCount");
719  static_assert(0U < TCount, "Incorrect instantiation");
720 
721  COMMS_ASSERT(TFromIdx <= idx);
722  COMMS_ASSERT(idx < TToIdx);
723  if (idx == TFromIdx) {
724  TupleSelectedTypeHelper<TFromIdx, TFromIdx + 1, 1U>::template exec<TTuple>(
725  idx, std::forward<TFunc>(func));
726  return;
727  }
728 
729  static const std::size_t MidIdx = TFromIdx + TCount / 2;
730  static_assert(MidIdx < TToIdx, "Internal error: bad calculation");
731  static_assert(TFromIdx <= MidIdx, "Internal error: bad calculation");
732  if (MidIdx <= idx) {
733  TupleSelectedTypeHelper<MidIdx, TToIdx, TToIdx - MidIdx>::template exec<TTuple>(
734  idx, std::forward<TFunc>(func));
735  return;
736  }
737 
738  TupleSelectedTypeHelper<TFromIdx, MidIdx, MidIdx - TFromIdx>::template exec<TTuple>(
739  idx, std::forward<TFunc>(func));
740  }
741 };
742 
743 template <std::size_t TFromIdx, std::size_t TToIdx>
744 class TupleSelectedTypeHelper<TFromIdx, TToIdx, 1U>
745 {

748 public:
