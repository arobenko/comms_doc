716 template <std::size_t TFromIdx, std::size_t TToIdx, std::size_t TCount>
717 class TupleSelectedTypeHelper
718 {
719  static_assert(TCount == (TToIdx - TFromIdx), "Internal error: Bad parameters");
720  static_assert(TFromIdx < TToIdx, "Internal error: Bad parameters");
721 public:
722  template <typename TTuple, typename TFunc>
723  static void exec(std::size_t idx, TFunc&& func)
724  {
725  using Tuple = typename std::decay<TTuple>::type;
726  static_assert(IsTuple<Tuple>::Value, "TTuple must be std::tuple");
727  static const std::size_t TupleSize = std::tuple_size<Tuple>::value;
728  static_assert(TCount <= TupleSize, "Incorrect TCount");
729  static_assert(0U < TCount, "Incorrect instantiation");
730 
731  COMMS_ASSERT(TFromIdx <= idx);
732  COMMS_ASSERT(idx < TToIdx);
733  if (idx == TFromIdx) {
734  TupleSelectedTypeHelper<TFromIdx, TFromIdx + 1, 1U>::template exec<TTuple>(
735  idx, std::forward<TFunc>(func));
736  return;
737  }
738 
739  static const std::size_t MidIdx = TFromIdx + TCount / 2;
740  static_assert(MidIdx < TToIdx, "Internal error: bad calculation");
741  static_assert(TFromIdx <= MidIdx, "Internal error: bad calculation");
742  if (MidIdx <= idx) {
743  TupleSelectedTypeHelper<MidIdx, TToIdx, TToIdx - MidIdx>::template exec<TTuple>(
744  idx, std::forward<TFunc>(func));
745  return;
746  }
747 
748  TupleSelectedTypeHelper<TFromIdx, MidIdx, MidIdx - TFromIdx>::template exec<TTuple>(
749  idx, std::forward<TFunc>(func));
750  }
751 };
