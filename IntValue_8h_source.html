349  template <typename TScaled>
350  void setScaledInternal(TScaled val, ScaleAsFpTag)
351  {
352  using DecayedType = typename std::decay<decltype(val)>::type;
353  auto epsilon = DecayedType(0);
354  if (ParsedOptions::ScalingRatio::num < ParsedOptions::ScalingRatio::den) {
355  epsilon = static_cast<DecayedType>(ParsedOptions::ScalingRatio::num) / static_cast<DecayedType>(ParsedOptions::ScalingRatio::den + 1);
356  }
357 
358  if (epsilon < DecayedType(0)) {
359  epsilon = -epsilon;
360  }
361 
362  if (val < DecayedType(0)) {
363  epsilon = -epsilon;
364  }
365 
366  BaseImpl::value() =
367  static_cast<ValueType>(
368  ((val + epsilon) * static_cast<DecayedType>(ParsedOptions::ScalingRatio::den)) / static_cast<DecayedType>(ParsedOptions::ScalingRatio::num));
369  }

376  std::intmax_t,
